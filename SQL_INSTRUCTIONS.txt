IMPORTANT: You must run the following SQL code in your Supabase Dashboard to fix the "not saving" issue.

1. Go to your Supabase Dashboard.
2. Click on the "SQL Editor" icon (left sidebar).
3. Click "New Query".
4. Paste the code below and click "Run".

---

-- 1. Fix Profiles Table Permissions (Crucial for saving changes)
ALTER TABLE public.profiles ENABLE ROW LEVEL SECURITY;

-- Drop old policies to avoid conflicts
DROP POLICY IF EXISTS "Public profiles are viewable by everyone" ON public.profiles;
DROP POLICY IF EXISTS "Users can insert own profile" ON public.profiles;
DROP POLICY IF EXISTS "Users can update own profile" ON public.profiles;
DROP POLICY IF EXISTS "Authenticated users can insert profiles" ON public.profiles;
DROP POLICY IF EXISTS "Authenticated users can update profiles" ON public.profiles;

-- Create robust policies that allow YOU to save your profile
CREATE POLICY "Public profiles are viewable by everyone" 
ON public.profiles FOR SELECT 
USING (true);

CREATE POLICY "Users can insert own profile" 
ON public.profiles FOR INSERT 
WITH CHECK (auth.uid() = id);

CREATE POLICY "Users can update own profile" 
ON public.profiles FOR UPDATE 
USING (auth.uid() = id);

-- 2. Grant necessary permissions
GRANT ALL ON public.profiles TO authenticated;
GRANT ALL ON public.profiles TO service_role;

-- 3. Fix Storage Permissions for 'Avatars' (Capitalized - used by @jeymuleya)
INSERT INTO storage.buckets (id, name, public)
VALUES ('Avatars', 'Avatars', true)
ON CONFLICT (id) DO UPDATE SET public = true;

DROP POLICY IF EXISTS "Public Access Avatars" ON storage.objects;
CREATE POLICY "Public Access Avatars" ON storage.objects FOR SELECT USING (bucket_id = 'Avatars');

DROP POLICY IF EXISTS "Authenticated users can upload Avatars" ON storage.objects;
CREATE POLICY "Authenticated users can upload Avatars" ON storage.objects FOR INSERT 
WITH CHECK (bucket_id = 'Avatars' AND auth.role() = 'authenticated');

DROP POLICY IF EXISTS "Users can update own Avatars" ON storage.objects;
CREATE POLICY "Users can update own Avatars" ON storage.objects FOR UPDATE
USING (bucket_id = 'Avatars' AND auth.uid() = owner);

-- 4. Fix Storage Permissions for 'avatars' (lowercase - fallback)
INSERT INTO storage.buckets (id, name, public)
VALUES ('avatars', 'avatars', true)
ON CONFLICT (id) DO UPDATE SET public = true;

DROP POLICY IF EXISTS "Public Access" ON storage.objects;
CREATE POLICY "Public Access" ON storage.objects FOR SELECT USING (bucket_id = 'avatars');

DROP POLICY IF EXISTS "Authenticated users can upload avatars" ON storage.objects;
CREATE POLICY "Authenticated users can upload avatars" ON storage.objects FOR INSERT 
WITH CHECK (bucket_id = 'avatars' AND auth.role() = 'authenticated');

DROP POLICY IF EXISTS "Users can update own avatars" ON storage.objects;
CREATE POLICY "Users can update own avatars" ON storage.objects FOR UPDATE
USING (bucket_id = 'avatars' AND auth.uid() = owner);
